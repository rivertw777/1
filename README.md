def signup():
    form = SignupForm()
    if form.validate_on_submit():
        select_user = User.query.filter_by().all()
        for user in select_user:
            if user.user_id == form.data.get('user_id'):  #입력받은 아이디가 이미 존재할시

        db.session.commit()

        return redirect('/login')
